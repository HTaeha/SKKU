CREATE TABLE USERINFO (USERID VARCHAR2(10), USERNAME VARCHAR2(20), AGE NUMBER, ISMALE NUMBER, PHONENUMBER VARCHAR2(13), EMAIL VARCHAR2(50), MONEY NUMBER, LOGINLOG VARCHAR2(20), SIGNUPLOG VARCHAR2(20), PRIMARY KEY(USERID));
CREATE TABLE TRAIN (TRAINID VARCHAR2(10),TRAINNAME VARCHAR2(20), TRAINTYPEID VARCHAR2(10), PRIMARY KEY(TRAINID), FOREIGN KEY(TRAINTYPEID) REFERENCES TRAINTYPE(TRAINTYPEID) );
CREATE TABLE TRAINTYPE (TRAINTYPEID VARCHAR2(10), TRAINTYPE VARCHAR2(10), PRIMARY KEY(TRAINTYPEID));
CREATE TABLE DESTINATION (DESTINATIONID VARCHAR(10), DESTINATIONNAME VARCHAR2(20), PRIMARY KEY(DESTINATIONID));
CREATE TABLE DESTINATIONPRICE (DESTINATIONPRICEID VARCHAR2(10), DISTANCE NUMBER, PRICE NUMBER, PRIMARY KEY(DESTINATIONPRICEID));
CREATE TABLE TRAINPRICE (TRAINPRICEID VARCHAR2(10), TRAINTYPEID VARCHAR2(10), PRICE NUMBER, PRIMARY KEY(TRAINPRICEID), FOREIGN KEY(TRAINTYPEID) REFERENCES TRAINTYPE(TRAINTYPEID));
CREATE TABLE SEATPRICE (SEATPRICEID VARCHAR2(10), SEATTYPEID VARCHAR2(10), PRICE NUMBER, PRIMARY KEY(SEATPRICEID), FOREIGN KEY(SEATTYPEID) REFERENCES SEATTYPE(SEATTYPEID));
CREATE TABLE DISTANCE(DISTANCEID VARCHAR2(10),DESTINATIONID1 VARCHAR2(10),DESTINATIONID2 VARCHAR2(10), DISTANCE NUMBER, PRIMARY KEY(DISTANCEID), FOREIGN KEY(DESTINATIONID1) REFERENCES DESTINATION(DESTINATIONID),FOREIGN KEY(DESTINATIONID2) REFERENCES DESTINATION(DESTINATIONID) );
CREATE TABLE LINE(LINEID VARCHAR2(10), STARTID VARCHAR2(10), DESTINATIONID VARCHAR2(10), STARTDATE VARCHAR2(20), STARTTIME VARCHAR2(20), TRAINID VARCHAR2(10), PRIMARY KEY(LINEID), FOREIGN KEY(STARTID) REFERENCES DESTINATION(DESTINATIONID), FOREIGN KEY(DESTINATIONID) REFERENCES DESTINATION(DESTINATIONID), FOREIGN KEY(TRAINID) REFERENCES TRAIN(TRAINID));
CREATE TABLE SEAT(SEATID VARCHAR2(10), LINEID VARCHAR2(10), BLOCK NUMBER, SEATNUM NUMBER, IS_FULL NUMBER, SEATTYPEID VARCHAR2(10), PRIMARY KEY(SEATID), FOREIGN KEY(LINEID) REFERENCES LINE(LINEID), FOREIGN KEY(SEATTYPEID) REFERENCES SEATTYPE(SEATTYPEID));
CREATE TABLE SEATTYPE(SEATTYPEID VARCHAR2(10), SEATTYPE VARCHAR2(10), PRIMARY KEY(SEATTYPEID));
CREATE TABLE RESERVATION(RESERVATIONID VARCHAR2(10), USERID VARCHAR2(10), LINEID VARCHAR2(10), SEATID VARCHAR2(10), FOREIGN KEY(USERID) REFERENCES USERINFO(USERID), FOREIGN KEY(LINEID) REFERENCES LINE(LINEID), FOREIGN KEY(SEATID) REFERENCES SEAT(SEATID));
